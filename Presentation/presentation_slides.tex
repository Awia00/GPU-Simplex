let simplex [n] [m] (A : [m][n]f32) (b : [m]f32)
  (c : [n]f32) (v : f32) =
  let e = entering_variable c
  let (_,_,_,v,_) = loop (A,b,c,v,e) while e != -1 do
    let l = leaving_variable A b e
    let (A,b,c,v) = unsafe pivot A b c v l e
    let e = entering_variable c
    in (A,b,c,v,e)
  in v
